## SPECIFIC PATTERNS TO DETECT

### Pattern 1: Missing Balance Check
```python
# DANGEROUS: No balance check
async def buy_item(item_id, price):
    await api.create_buy_order(item_id, price)

# SAFE: Check balance first
async def buy_item(item_id, price):
    balance = await api.get_balance()
    effective = balance.available - balance.reserved
    if effective < price:
        raise InsufficientBalance(f"Need {price}, have {effective}")
    await api.create_buy_order(item_id, price)
```

### Pattern 2: Fee Miscalculation
```python
# WRONG: Only considering buy fee
margin = (sell_price - buy_price - buy_fee) / buy_price

# WRONG: Using wrong fee percentage
margin = (sell_price - buy_price - buy_price * 0.05) / buy_price

# CORRECT: Both fees, correct percentages
buy_fee = buy_price * 0.05  # 5% buy fee
sell_fee = sell_price * 0.10  # 10% sell fee
margin = (sell_price - buy_price - buy_fee - sell_fee) / buy_price
```

### Pattern 3: Incorrect Kelly Formula
```python
# WRONG: Full Kelly (too aggressive)
kelly_f = (win_rate * avg_win - (1 - win_rate) * avg_loss) / avg_win

# WRONG: No floor/ceiling
position_size = balance * kelly_f

# CORRECT: Half Kelly with bounds
kelly_f = (win_rate * avg_win - (1 - win_rate) * avg_loss) / avg_win
half_kelly = kelly_f * 0.5
position_size = balance * max(0.03, min(0.10, half_kelly))  # 3% floor, 10% cap
```

### Pattern 4: Missing Drawdown Freeze
```python
# DANGEROUS: No drawdown protection
async def execute_trade(trade):
    await api.create_order(trade)

# SAFE: Check drawdown
async def execute_trade(trade):
    balance = await api.get_balance()
    peak = load_peak_balance()
    if balance < peak * 0.85:
        raise DrawdownFreeze(f"Balance {balance} < 85% of peak {peak}")
    await api.create_order(trade)
```

### Pattern 5: No Idempotency
```python
# DANGEROUS: No idempotency key
async def create_order(item_id, price):
    return await api.create_buy_order(item_id, price)

# SAFE: Unique idempotency key
async def create_order(item_id, price):
    client_order_id = f"{item_id}_{int(time.time())}_{hash(item_id)}"
    return await api.create_buy_order(item_id, price, client_order_id=client_order_id)
```

### Pattern 6: Race Condition in Trading
```python
# DANGEROUS: Two tasks can buy same item
async def scan_and_buy():
    items = await scan_market()
    for item in items:
        asyncio.create_task(buy_item(item))  # No coordination

# SAFE: Serialize purchases
async def scan_and_buy():
    items = await scan_market()
    async with trade_lock:
        for item in items:
            await buy_item(item)
```

### Pattern 7: Stale Price Data
```python
# DANGEROUS: Using old price
price = get_cached_price(item_id)  # Could be hours old
if price < threshold:
    buy(item_id, price)

# SAFE: Check price freshness
price = get_cached_price(item_id)
if time.time() - price.timestamp > 300:  # 5 minutes
    price = await fetch_fresh_price(item_id)
if price.value < threshold:
    buy(item_id, price.value)
```

### Pattern 8: Missing Slippage Protection
```python
# DANGEROUS: No re-verification
async def execute_trade(item_id, expected_price):
    await api.create_buy_order(item_id, expected_price)

# SAFE: Re-verify before execution
async def execute_trade(item_id, expected_price, max_slippage=0.02):
    current_price = await api.get_price(item_id)
    slippage = abs(current_price - expected_price) / expected_price
    if slippage > max_slippage:
        raise SlippageExceeded(f"Price moved {slippage:.2%}")
    await api.create_buy_order(item_id, current_price)
```
